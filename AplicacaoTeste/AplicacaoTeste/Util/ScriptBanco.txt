CREATE DATABASE BDContato;
GO

USE BDContato;
GO

CREATE TABLE PESSOA(
ID_PESSOA INT PRIMARY KEY IDENTITY,
CEP VARCHAR(11),
LOGRADOURO VARCHAR(30),
NUMERO INT,
COMPLEMENTO VARCHAR(30),
BAIRRO VARCHAR(30),
CIDADE VARCHAR(30),
UF VARCHAR(30),
);

CREATE TABLE PESSOAJURIDICA(
ID_PESSOAJURIDICA INT PRIMARY KEY IDENTITY,
ID_PESSOA INT,
CNPJ VARCHAR(18),
RAZAOSOCIAL VARCHAR(30),
NOMEFANTASIA VARCHAR(30),
CONSTRAINT FK_PESSOAJURIDICA FOREIGN KEY (ID_PESSOA) REFERENCES PESSOA(ID_PESSOA)
);


CREATE TABLE PESSOAFISICA(
ID_PESSOAFISICA INT PRIMARY KEY IDENTITY,
ID_PESSOA INT,
CPF VARCHAR(14),
DATA_NASCIMENTO DATETIME,
NOME VARCHAR(30),
SOBRENOME VARCHAR(30),
CONSTRAINT FK_PESSOAFISICA FOREIGN KEY (ID_PESSOA) REFERENCES PESSOA(ID_PESSOA)
);
